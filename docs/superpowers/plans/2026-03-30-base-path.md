# Base Path Support Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Allow Cetacean to serve under a configurable URL prefix (e.g., `/cetacean/`) via a strip-then-delegate wrapper middleware.

**Architecture:** A new outermost middleware strips the configured base path from inbound requests and stores it in context. Handlers use a `basePath(ctx)` helper to prepend the prefix to outbound URLs (`@id`, Link headers, error URIs). The SPA handler injects `<base>` and `<link rel="canonical">` into `index.html` at startup. The frontend reads `document.baseURI` for API/SSE/router prefixing.

**Tech Stack:** Go stdlib `net/http`, React Router `basename`, Vite (no build-time changes)

**Spec:** `docs/superpowers/specs/2026-03-30-base-path-design.md`

```go
package config

import (
	"fmt"
	"strings"
)

// NormalizeBasePath cleans a base path to canonical form: leading slash,
// no trailing slash. Empty string or "/" both normalize to "" (root).
func NormalizeBasePath(s string) string {
	s = strings.Trim(s, "/")
	if s == "" {
		return ""
	}
	return "/" + s
}

// ValidateBasePath checks that a normalized base path is well-formed:
// no query strings, fragments, or double slashes.
func ValidateBasePath(bp string) error {
	if bp == "" {
		return nil
	}
	if strings.Contains(bp, "?") {
		return fmt.Errorf("base path must not contain query string: %q", bp)
	}
	if strings.Contains(bp, "#") {
		return fmt.Errorf("base path must not contain fragment: %q", bp)
	}
	if strings.Contains(bp, "//") {
		return fmt.Errorf("base path must not contain double slashes: %q", bp)
	}
	return nil
}
```

```go
package api

import (
	"context"
	"net/http"
	"strings"
)

type basePathCtxKey struct{}

var basePathKey = basePathCtxKey{}

// BasePathFromContext returns the configured base path from the request context.
func BasePathFromContext(ctx context.Context) string {
	if bp, ok := ctx.Value(basePathKey).(string); ok {
		return bp
	}
	return ""
}

// absPath prepends the base path from context to a path string.
func absPath(ctx context.Context, path string) string {
	bp := BasePathFromContext(ctx)
	if bp == "" {
		return path
	}
	return bp + path
}

// basePathMiddleware strips the configured base path from inbound requests,
// stores it in context for outbound URL generation, and handles trailing
// slash canonicalization.
func basePathMiddleware(basePath string, next http.Handler) http.Handler {
	if basePath == "" {
		return next
	}

	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		path := r.URL.Path

		// Must start with base path.
		if !strings.HasPrefix(path, basePath) {
			http.NotFound(w, r)
			return
		}

		// Strip the base path.
		stripped := strings.TrimPrefix(path, basePath)

		// Base path root: /cetacean or /cetacean/
		if stripped == "" || stripped == "/" {
			r.URL.Path = "/"
			ctx := context.WithValue(r.Context(), basePathKey, basePath)
			next.ServeHTTP(w, r.WithContext(ctx))
			return
		}

		// Trailing slash redirect (except root which is handled above).
		if len(stripped) > 1 && strings.HasSuffix(stripped, "/") {
			target := basePath + strings.TrimSuffix(stripped, "/")
			if r.URL.RawQuery != "" {
				target += "?" + r.URL.RawQuery
			}
			http.Redirect(w, r, target, http.StatusMovedPermanently)
			return
		}

		r.URL.Path = stripped
		ctx := context.WithValue(r.Context(), basePathKey, basePath)
		next.ServeHTTP(w, r.WithContext(ctx))
	})
}
```

```go
func writePaginationLinks(w http.ResponseWriter, r *http.Request, total, limit, offset int) {
	buildLink := func(newOffset int) string {

		q.Set("limit", strconv.Itoa(limit))
```

```go
package api

import (
	"io/fs"
	"net/http"
	"strings"
)

// NewSPAHandler serves the embedded frontend filesystem with index.html
// fallback for client-side routing. When basePath is non-empty, it injects
// <base> and <link rel="canonical"> tags into index.html at startup.
func NewSPAHandler(fsys fs.FS, basePath string) http.Handler {
	fileServer := http.FileServer(http.FS(fsys))

	// Read and prepare index.html with base path injection.
	indexBytes, _ := fs.ReadFile(fsys, "index.html")
	indexHTML := string(indexBytes)

	baseHref := "/"
	if basePath != "" {
		baseHref = basePath + "/"
	}

	injection := `<base href="` + baseHref + `">` + "\n" +
		`    <link rel="canonical" href="` + baseHref + `">`

	indexHTML = strings.Replace(indexHTML, "<head>", "<head>\n    "+injection, 1)
	preparedIndex := []byte(indexHTML)

	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		path := strings.TrimPrefix(r.URL.Path, "/")
		if path == "" {
			path = "index.html"
		}

		f, err := fsys.Open(path)
		if err != nil {
			// Fall back to prepared index.html for client-side routing.
			w.Header().Set("Content-Type", "text/html; charset=utf-8")
			w.Write(preparedIndex)
			return
		}
		_ = f.Close()

		// For index.html itself, serve the prepared version.
		if path == "index.html" {
			w.Header().Set("Content-Type", "text/html; charset=utf-8")
			w.Write(preparedIndex)
			return
		}

		// All other static files served as-is.
		fileServer.ServeHTTP(w, r)
	})
}
```

```typescript
/**
 * Reads the base path from the <base> tag injected by the Go server.
 * Returns "" when running at root, or "/cetacean" (no trailing slash)
 * when running under a sub-path.
 */
function detectBasePath(): string {
  try {
    const base = new URL(document.baseURI);
    const path = base.pathname.replace(/\/+$/, "");

    return path;
  } catch {
    return "";
  }
}

const cachedBasePath = detectBasePath();

/**
 * The configured base path, e.g. "" or "/cetacean".
 * No trailing slash.
 */
export const basePath = cachedBasePath;

/**
 * Prepends the base path to an absolute path.
 * apiPath("/nodes") → "/cetacean/nodes" (or "/nodes" at root).
 */
export function apiPath(path: string): string {
  if (!cachedBasePath) {
    return path;
  }

  return cachedBasePath + path;
}
```
