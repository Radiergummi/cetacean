> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Architecture:** Astro content collection reads Markdown from `../docs/`, renders with a three-column layout (sidebar, content, TOC). Design tokens imported from Cetacean's CSS. GitHub Actions builds and deploys.

**Tech Stack:** Astro 5, Tailwind CSS v4, `@fontsource-variable/geist`, Shiki syntax highlighting

**Spec:** `docs/superpowers/specs/2026-04-02-docs-site-design.md`

```astro
  </svg>
  <svg id="theme-icon-dark" class="hidden size-4" fill="none" viewBox="0 0 24 24" stroke="currentColor" stroke-width="2">
    <path d="M21 12.79A9 9 0 1 1 11.21 3 7 7 0 0 0 21 12.79z" />
  </svg>
  <svg id="theme-icon-system" class="hidden size-4" fill="none" viewBox="0 0 24 24" stroke="currentColor" stroke-width="2">
    <rect x="2" y="3" width="20" height="14" rx="2" ry="2" />
    <path d="M8 21h8m-4-4v4" />
  </svg>
</button>

<script is:inline>
  (function() {
    const themes = ["light", "dark", "system"];
    function getStored() { return localStorage.getItem("theme") || "system"; }
    function apply(theme) {
      const dark = theme === "dark" || (theme === "system" && window.matchMedia("(prefers-color-scheme: dark)").matches);
      document.documentElement.classList.toggle("dark", dark);
      document.getElementById("theme-icon-light").classList.toggle("hidden", theme !== "light");
      document.getElementById("theme-icon-dark").classList.toggle("hidden", theme !== "dark");
      document.getElementById("theme-icon-system").classList.toggle("hidden", theme !== "system");
    }
    apply(getStored());
    document.getElementById("theme-toggle").addEventListener("click", function() {
      const current = getStored();
      const next = themes[(themes.indexOf(current) + 1) % themes.length];
      localStorage.setItem("theme", next);
      apply(next);
    });
    window.matchMedia("(prefers-color-scheme: dark)").addEventListener("change", function() {
      if (getStored() === "system") apply("system");
    });
  })();
</script>
```

### Task 6: Table of Contents component

**Files:**
- Create: `website/src/components/TableOfContents.astro`

- [ ] **Step 1: Create `website/src/components/TableOfContents.astro`**

Receives headings from the rendered Markdown and displays them as a sticky right-side nav. Active heading highlighting uses a small client-side IntersectionObserver.

```astro
---
interface Heading {
  depth: number;
  slug: string;
  text: string;
}

interface Props {
  headings: Heading[];
}

const { headings } = Astro.props;
const filtered = headings.filter((h) => h.depth >= 2 && h.depth <= 3);
---
{filtered.length > 0 && (
  <aside class="hidden xl:block w-44 shrink-0 sticky top-14 h-[calc(100vh-3.5rem)] overflow-y-auto py-6 px-4">
    <h3 class="text-xs font-semibold uppercase tracking-wider text-muted-foreground mb-3">
      On this page
    </h3>
    <ul id="toc-list" class="space-y-1 text-sm">
      {filtered.map((heading) => (
        <li>
          <a
            href={`#${heading.slug}`}
            class:list={[
              "block py-0.5 text-muted-foreground hover:text-foreground transition-colors",
              heading.depth === 3 && "pl-3",
            ]}
            data-heading={heading.slug}
          >
            {heading.text}
          </a>
        </li>
      ))}
    </ul>
  </aside>
)}

<script is:inline>
  (function() {
    const links = document.querySelectorAll("#toc-list a[data-heading]");
    if (!links.length) return;
    const headings = Array.from(links).map(function(link) {
      return { link: link, target: document.getElementById(link.dataset.heading) };
    }).filter(function(h) { return h.target; });
    var current = null;
    var observer = new IntersectionObserver(function(entries) {
      entries.forEach(function(entry) {
        if (entry.isIntersecting) {
          if (current) current.classList.remove("text-foreground", "font-medium");
          var match = headings.find(function(h) { return h.target === entry.target; });
          if (match) {
            match.link.classList.add("text-foreground", "font-medium");
            current = match.link;
          }
        }
      });
    }, { rootMargin: "-80px 0px -80% 0px" });
    headings.forEach(function(h) { observer.observe(h.target); });
  })();
</script>
```


Actually, Astro doesn't have a built-in `renderMarkdown` function for arbitrary strings at build time. A simpler approach: use a `<Content />` component from a manually created collection entry, or use a Markdown library. The simplest: use `marked` to render the changelog.

```bash
cd website && npm install marked
```

Then:

```astro
---
import DocsLayout from "../layouts/DocsLayout.astro";
import fs from "node:fs";
import path from "node:path";
import { marked } from "marked";

const changelogPath = path.resolve("../CHANGELOG.md");
const raw = fs.readFileSync(changelogPath, "utf-8");
const html = await marked(raw);
---
<DocsLayout
  title="Changelog"
  description="Release history"
  slug="changelog"
  headings={[]}
>
  <Fragment set:html={html} />
</DocsLayout>
```

### Task 11: Code block copy button

**Files:**
- Modify: `website/src/layouts/DocsLayout.astro`

- [ ] **Step 1: Add copy button script to DocsLayout**

Add before the closing `</body>` tag in `DocsLayout.astro`:

```html
<script is:inline>
  document.querySelectorAll(".prose pre").forEach(function(pre) {
    var button = document.createElement("button");
    button.className = "copy-button";
    button.textContent = "Copy";
    button.addEventListener("click", function() {
      var code = pre.querySelector("code");
      navigator.clipboard.writeText(code ? code.textContent : pre.textContent).then(function() {
        button.textContent = "Copied!";
        setTimeout(function() { button.textContent = "Copy"; }, 2000);
      });
    });
    pre.style.position = "relative";
    pre.appendChild(button);
  });
</script>
```

```yaml
name: Deploy docs to GitHub Pages

on:
  push:
    branches: [main]
    paths:
      - "docs/**"
      - "website/**"
      - "CHANGELOG.md"
      - ".github/workflows/pages.yml"
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

concurrency:
  group: pages
  cancel-in-progress: true

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: 22

      - name: Install and build
        working-directory: website
        run: |
          npm ci
          npm run build

      - uses: actions/configure-pages@v5

      - uses: actions/upload-pages-artifact@v3
        with:
          path: website/dist

  deploy:
    runs-on: ubuntu-latest
    needs: build
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    steps:
      - id: deployment
        uses: actions/deploy-pages@v4
```
