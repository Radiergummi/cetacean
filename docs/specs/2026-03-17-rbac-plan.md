# RBAC Authorization Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add grant-based RBAC authorization so users can control per-resource visibility and permissions via policy files and provider-specific grant sources.

**Architecture:** A standalone `internal/acl` package owns policy parsing, matching, and evaluation. Handlers use `acl.Filter()` for lists and `acl.Can()` for details. SSE match functions are wrapped with authorization checks. Grant sources are provider-pluggable via a `GrantSource` interface. The frontend reads a permissions summary from `/profile` and hides inaccessible UI elements.

**Tech Stack:** Go stdlib, `path.Match` for glob matching, `fsnotify` for hot reload, React hooks for frontend permission checks

**Spec:** `docs/specs/2026-03-17-rbac-design.md`

```go
// internal/acl/policy.go
package acl

import (
	"fmt"
	"os"
	"path/filepath"
	"strings"

	json "github.com/goccy/go-json"
	"github.com/pelletier/go-toml/v2"
	"gopkg.in/yaml.v3"
)

type Grant struct {
	Resources   []string `json:"resources" yaml:"resources" toml:"resources"`
	Audience    []string `json:"audience" yaml:"audience" toml:"audience"`
	Permissions []string `json:"permissions" yaml:"permissions" toml:"permissions"`
}

type Policy struct {
	Grants []Grant `json:"grants" yaml:"grants" toml:"grants"`
}

var validResourceTypes = map[string]bool{
	"service": true, "stack": true, "node": true, "task": true,
	"config": true, "secret": true, "network": true, "volume": true,
}

var validAudienceKinds = map[string]bool{
	"user": true, "group": true,
}

var validPermissions = map[string]bool{
	"read": true, "write": true,
}

// ParsePolicy parses a policy from bytes. ext is the file extension hint
// (e.g. ".json", ".yaml", ".toml"). If empty, tries JSON → TOML → YAML.
func ParsePolicy(data []byte, ext string) (*Policy, error) {
	ext = strings.ToLower(ext)
	var p Policy
	var err error

	switch ext {
	case ".json":
		err = json.Unmarshal(data, &p)
	case ".yaml", ".yml":
		err = yaml.Unmarshal(data, &p)
	case ".toml":
		err = toml.Unmarshal(data, &p)
	default:
		// Auto-detect: JSON → TOML → YAML
		if err = json.Unmarshal(data, &p); err == nil {
			return validated(&p)
		}
		if err = toml.Unmarshal(data, &p); err == nil {
			return validated(&p)
		}
		err = yaml.Unmarshal(data, &p)
	}
	if err != nil {
		return nil, fmt.Errorf("parsing policy: %w", err)
	}
	return validated(&p)
}

func validated(p *Policy) (*Policy, error) {
	if err := p.Validate(); err != nil {
		return nil, err
	}
	return p, nil
}

// Validate checks that all grants use valid resource types, audience kinds,
// and permissions.
func (p *Policy) Validate() error {
	for i, g := range p.Grants {
		for _, r := range g.Resources {
			if r == "*" {
				continue
			}
			typ, _, ok := strings.Cut(r, ":")
			if !ok {
				return fmt.Errorf("grant %d: invalid resource expression %q (expected type:pattern)", i, r)
			}
			if !validResourceTypes[typ] {
				return fmt.Errorf("grant %d: unknown resource type %q", i, typ)
			}
		}
		for _, a := range g.Audience {
			if a == "*" {
				continue
			}
			kind, _, ok := strings.Cut(a, ":")
			if !ok {
				return fmt.Errorf("grant %d: invalid audience expression %q (expected kind:pattern)", i, a)
			}
			if !validAudienceKinds[kind] {
				return fmt.Errorf("grant %d: unknown audience kind %q", i, kind)
			}
		}
		for _, perm := range g.Permissions {
			if !validPermissions[perm] {
				return fmt.Errorf("grant %d: unknown permission %q", i, perm)
			}
		}
	}
	return nil
}

// ParsePolicyFile reads a policy file and detects format from extension.
func ParsePolicyFile(path string) (*Policy, error) {
	// Import os in actual implementation
	data, err := os.ReadFile(filepath.Clean(path))
	if err != nil {
		return nil, fmt.Errorf("reading policy file: %w", err)
	}
	return ParsePolicy(data, filepath.Ext(path))
}
```

```go
// internal/acl/match.go
package acl

import (
	"path"
	"strings"

	"github.com/radiergummi/cetacean/internal/auth"
)

// matchResource checks if a resource expression matches a resource string.
// pattern is "type:glob" or "*" (matches everything).
// resource is "type:name", e.g. "service:webapp-api".
func matchResource(pattern, resource string) bool {
	if pattern == "*" {
		return true
	}
	pType, pGlob, ok := strings.Cut(pattern, ":")
	if !ok {
		return false
	}
	rType, rName, ok := strings.Cut(resource, ":")
	if !ok {
		return false
	}
	if pType != rType {
		return false
	}
	matched, _ := path.Match(pGlob, rName)
	return matched
}

// matchAudience checks if an audience expression matches an identity.
// pattern is "user:glob", "group:glob", or "*".
// user: matches against both Subject and Email (union).
func matchAudience(pattern string, id *auth.Identity) bool {
	if pattern == "*" {
		return true
	}
	kind, glob, ok := strings.Cut(pattern, ":")
	if !ok {
		return false
	}
	switch kind {
	case "user":
		if matched, _ := path.Match(glob, id.Subject); matched {
			return true
		}
		if id.Email != "" {
			if matched, _ := path.Match(glob, id.Email); matched {
				return true
			}
		}
		return false
	case "group":
		for _, g := range id.Groups {
			if matched, _ := path.Match(glob, g); matched {
				return true
			}
		}
		return false
	default:
		return false
	}
}
```

```go
// Evaluator checks authorization grants. Thread-safe.
// A nil policy means no authorization (allow all). A non-nil policy with
// no matching grants means deny.
type Evaluator struct {
	policy   atomic.Pointer[Policy]
	source   GrantSource
	resolver ResourceResolver
}

// NewEvaluator creates an evaluator with optional grant source and resolver.
func NewEvaluator(source GrantSource, resolver ResourceResolver) *Evaluator {
	return &Evaluator{source: source, resolver: resolver}
}

// SetPolicy atomically replaces the current policy.
func (e *Evaluator) SetPolicy(p *Policy) {
	e.policy.Store(p)
}

// Can checks if the identity has the given permission on the resource.
// resource is "type:name", e.g. "service:webapp-api".
func (e *Evaluator) Can(id *auth.Identity, permission string, resource string) bool {
	p := e.policy.Load()
	if p == nil {
		// No policy configured — allow everything
		return true
	}
	if id == nil {
		return false
	}

	grants := e.grantsFor(id, p)
	return e.checkGrants(grants, permission, resource)
}

// HasAnyGrant returns true if the identity matches at least one grant.
// Used for cluster-wide endpoints that require any grant to access.
func (e *Evaluator) HasAnyGrant(id *auth.Identity) bool {
	p := e.policy.Load()
	if p == nil {
		return true
	}
	if id == nil {
		return false
	}
	grants := e.grantsFor(id, p)
	return len(grants) > 0
}

// Filter returns only items the identity can access with the given permission.
func Filter[T any](e *Evaluator, id *auth.Identity, permission string, items []T, resourceFunc func(T) string) []T {
	p := e.policy.Load()
	if p == nil {
		return items
	}
	if id == nil {
		return nil
	}
	grants := e.grantsFor(id, p)
	if len(grants) == 0 {
		return nil
	}
	result := make([]T, 0, len(items))
	for _, item := range items {
		if e.checkGrants(grants, permission, resourceFunc(item)) {
			result = append(result, item)
		}
	}
	return result
}

// PermissionsFor returns a map of resource patterns to permissions for the identity.
// Used by /profile to tell the frontend what the user can access.
func (e *Evaluator) PermissionsFor(id *auth.Identity) map[string][]string {
	p := e.policy.Load()
	if p == nil {
		return nil
	}
	if id == nil {
		return nil
	}
	grants := e.grantsFor(id, p)
	if len(grants) == 0 {
		return map[string][]string{}
	}
	perms := make(map[string][]string)
	for _, g := range grants {
		for _, r := range g.Resources {
			perms[r] = appendUnique(perms[r], g.Permissions...)
		}
	}
	return perms
}

// grantsFor collects all grants that match the identity from both
// the file policy and the provider grant source.
func (e *Evaluator) grantsFor(id *auth.Identity, p *Policy) []Grant {
	var grants []Grant
	for _, g := range p.Grants {
		if audienceMatches(g.Audience, id) {
			grants = append(grants, g)
		}
	}
	if e.source != nil {
		grants = append(grants, e.source.GrantsFor(id)...)
	}
	return grants
}

// checkGrants checks if any of the grants allow the permission on the resource.
func (e *Evaluator) checkGrants(grants []Grant, permission, resource string) bool {
	for _, g := range grants {
		if !hasPermission(g.Permissions, permission) {
			continue
		}
		for _, r := range g.Resources {
			if matchResource(r, resource) {
				return true
			}
		}
		// Check stack membership if resolver is available
		if e.resolver != nil {
			if e.checkStackGrant(g, resource) {
				return true
			}
			if e.checkTaskInheritance(g, permission, resource) {
				return true
			}
		}
	}
	return false
}

// checkStackGrant checks if a grant's stack resources cover the given resource.
func (e *Evaluator) checkStackGrant(g Grant, resource string) bool {
	rType, rID, ok := strings.Cut(resource, ":")
	if !ok {
		return false
	}
	stackName := e.resolver.StackOf(rType, rID)
	if stackName == "" {
		return false
	}
	for _, r := range g.Resources {
		if matchResource(r, "stack:"+stackName) {
			return true
		}
	}
	return false
}

// checkTaskInheritance checks if a task inherits access from its parent service.
func (e *Evaluator) checkTaskInheritance(g Grant, permission, resource string) bool {
	rType, rID, ok := strings.Cut(resource, ":")
	if !ok || rType != "task" {
		return false
	}
	serviceName := e.resolver.ServiceOfTask(rID)
	if serviceName == "" {
		return false
	}
	// Check direct service match
	for _, r := range g.Resources {
		if matchResource(r, "service:"+serviceName) {
			return true
		}
	}
	// Check stack match via service
	stackName := e.resolver.StackOf("service", serviceName)
	if stackName == "" {
		return false
	}
	for _, r := range g.Resources {
		if matchResource(r, "stack:"+stackName) {
			return true
		}
	}
	return false
}

func audienceMatches(audience []string, id *auth.Identity) bool {
	for _, a := range audience {
		if matchAudience(a, id) {
			return true
		}
	}
	return false
}

func hasPermission(granted []string, requested string) bool {
	for _, p := range granted {
		if p == requested {
			return true
		}
		// write implies read
		if p == "write" && requested == "read" {
			return true
		}
	}
	return false
}

func appendUnique(slice []string, items ...string) []string {
	seen := make(map[string]bool, len(slice))
	for _, s := range slice {
		seen[s] = true
	}
	for _, item := range items {
		if !seen[item] {
			slice = append(slice, item)
			seen[item] = true
		}
	}
	return slice
}
```

```go
	}
```

```go
// StackOf returns the stack name for a resource, identified by name.
// Returns "" if the resource doesn't exist or isn't in a stack.
func (c *Cache) StackOf(resourceType, name string) string {
	c.mu.RLock()
	defer c.mu.RUnlock()
	switch resourceType {
	case "service":
		for _, s := range c.services {
			if s.Spec.Name == name {
				return s.Spec.Labels[stackLabel]
			}
		}
	case "config":
		for _, cfg := range c.configs {
			if cfg.Spec.Name == name {
				return cfg.Spec.Labels[stackLabel]
			}
		}
	case "secret":
		for _, s := range c.secrets {
			if s.Spec.Name == name {
				return s.Spec.Labels[stackLabel]
			}
		}
	case "network":
		for _, n := range c.networks {
			if n.Name == name {
				return n.Labels[stackLabel]
			}
		}
	case "volume":
		for _, v := range c.volumes {
			if v.Name == name {
				return v.Labels[stackLabel]
			}
		}
	}
	return ""
}

// ServiceOfTask returns the service name for a task, or "" if unknown.
func (c *Cache) ServiceOfTask(taskID string) string {
	c.mu.RLock()
	defer c.mu.RUnlock()
	t, ok := c.tasks[taskID]
	if !ok {
		return ""
	}
	s, ok := c.services[t.ServiceID]
	if !ok {
		return ""
	}
	return s.Spec.Name
}
```

```go
// internal/acl/reload.go
package acl

import (
	"log/slog"
	"os"
	"path/filepath"
	"time"

	"github.com/fsnotify/fsnotify"
)

// WatchPolicyFile loads the policy file and watches for changes.
// Returns a stop function and any error from initial setup.
// The evaluator's policy is set on initial load and updated on changes.
func WatchPolicyFile(e *Evaluator, path string) (stop func(), err error) {
	// Initial load
	if err := loadAndSet(e, path); err != nil {
		return nil, err
	}

	watcher, err := fsnotify.NewWatcher()
	if err != nil {
		return nil, err
	}

	// Watch the directory (not the file) to handle editors that
	// write to a temp file and rename (atomic save).
	dir := filepath.Dir(path)
	if err := watcher.Add(dir); err != nil {
		watcher.Close()
		return nil, err
	}

	done := make(chan struct{})
	go func() {
		defer watcher.Close()
		var debounce *time.Timer
		base := filepath.Base(path)
		for {
			select {
			case event, ok := <-watcher.Events:
				if !ok {
					return
				}
				if filepath.Base(event.Name) != base {
					continue
				}
				if event.Op&(fsnotify.Write|fsnotify.Create) == 0 {
					continue
				}
				if debounce != nil {
					debounce.Stop()
				}
				debounce = time.AfterFunc(200*time.Millisecond, func() {
					if err := loadAndSet(e, path); err != nil {
						slog.Error("failed to reload ACL policy", "path", path, "error", err)
					} else {
						slog.Info("ACL policy reloaded", "path", path)
					}
				})
			case err, ok := <-watcher.Errors:
				if !ok {
					return
				}
				slog.Error("ACL policy watcher error", "error", err)
			case <-done:
				return
			}
		}
	}()

	return func() { close(done) }, nil
}

func loadAndSet(e *Evaluator, path string) error {
	data, err := os.ReadFile(filepath.Clean(path))
	if err != nil {
		return err
	}
	p, err := ParsePolicy(data, filepath.Ext(path))
	if err != nil {
		return err
	}
	e.SetPolicy(p)
	return nil
}
```

```typescript
// frontend/src/hooks/useCanAccess.ts
import { useAuth } from "@/hooks/useAuth";

/**
 * Checks if the current user has the given permission on the resource.
 * Uses simple glob matching (only * wildcard) against the permissions
 * map from /profile.
 *
 * Returns true if:
 * - No permissions map (no ACL configured, everything allowed)
 * - Any grant pattern matches the resource with the requested permission
 */
export function useCanAccess(resource: string, permission: string): boolean {
  const { identity } = useAuth();
  if (!identity) return false;
  if (!identity.permissions) return true; // no ACL = full access

  for (const [pattern, perms] of Object.entries(identity.permissions)) {
    if (!perms.includes(permission) && !(permission === "read" && perms.includes("write"))) {
      continue;
    }
    if (matchGlob(pattern, resource)) {
      return true;
    }
  }
  return false;
}

function matchGlob(pattern: string, value: string): boolean {
  if (pattern === "*") return true;

  const [pType, pGlob] = splitOnce(pattern, ":");
  const [vType, vName] = splitOnce(value, ":");
  if (!pGlob || !vName) return false;
  if (pType !== vType) return false;

  return simpleGlob(pGlob, vName);
}

function simpleGlob(pattern: string, value: string): boolean {
  // Simple * matching only (no ? support client-side)
  const parts = pattern.split("*");
  if (parts.length === 1) return pattern === value;

  let pos = 0;
  for (let i = 0; i < parts.length; i++) {
    const part = parts[i];
    if (i === 0 && part !== "") {
      if (!value.startsWith(part)) return false;
      pos = part.length;
    } else if (i === parts.length - 1 && part !== "") {
      if (!value.endsWith(part)) return false;
    } else if (part !== "") {
      const idx = value.indexOf(part, pos);
      if (idx === -1) return false;
      pos = idx + part.length;
    }
  }
  return true;
}

function splitOnce(s: string, sep: string): [string, string | undefined] {
  const i = s.indexOf(sep);
  if (i === -1) return [s, undefined];
  return [s.substring(0, i), s.substring(i + 1)];
}
```
